// Particles
const pContainer = document.getElementById('particles');
for (let i = 0; i < 40; i++) {
  const p = document.createElement('div');
  p.className = 'particle';
  const colors = ['#00f5d4','#8b5cf6','#f59e0b','#fff'];
  p.style.cssText = `
    left: ${Math.random()*100}%;
    animation-duration: ${6 + Math.random()*14}s;
    animation-delay: ${Math.random()*10}s;
    --dx: ${(Math.random()-0.5)*120}px;
    background: ${colors[Math.floor(Math.random()*colors.length)]};
    width: ${1+Math.random()*2}px;
    height: ${1+Math.random()*2}px;
    opacity: 0;
  `;
  pContainer.appendChild(p);
}

// Progress counter
const pct = document.getElementById('pct');
const bar = document.getElementById('bar');
const steps = [
  { el: document.getElementById('s1'), at: 20 },
  { el: document.getElementById('s2'), at: 50 },
  { el: document.getElementById('s3'), at: 75 },
  { el: document.getElementById('s4'), at: 90 },
];

// Progress targets from the CSS animation keyframes
const targets = [
  { t: 0,    v: 0  },
  { t: 1600, v: 24 },
  { t: 3600, v: 51 },
  { t: 5200, v: 68 },
  { t: 6400, v: 79 },
  { t: 7200, v: 88 },
  { t: 8000, v: 94 },
];

let current = 0;
let stepsDone = new Set();
let activeSet = new Set();

function lerp(a, b, t) { return a + (b-a)*t; }
function getTarget(elapsed) {
  for (let i = targets.length-1; i >= 0; i--) {
    if (elapsed >= targets[i].t) {
      const next = targets[i+1];
      if (!next) return targets[i].v;
      const progress = (elapsed - targets[i].t) / (next.t - targets[i].t);
      return lerp(targets[i].v, next.v, Math.min(progress,1));
    }
  }
  return 0;
}

const start = Date.now();
function tick() {
  const elapsed = Date.now() - start;
  const val = Math.round(getTarget(elapsed));
  current = val;
  pct.textContent = val + '%';

  // Update steps
  steps.forEach(s => {
    if (val >= s.at && !stepsDone.has(s.el)) {
      if (activeSet.has(s.el)) { activeSet.delete(s.el); }
      s.el.className = 'step done';
      stepsDone.add(s.el);
    } else if (val >= s.at - 10 && val < s.at && !stepsDone.has(s.el)) {
      if (!activeSet.has(s.el)) {
        s.el.className = 'step active';
        activeSet.add(s.el);
      }
    }
  });

  if (elapsed < 8200) requestAnimationFrame(tick);
}
requestAnimationFrame(tick);

// Timer countdown
let secs = 150;
const timerEl = document.getElementById('timer');
const countdown = setInterval(() => {
  if (secs <= 0) { clearInterval(countdown); return; }
  secs--;
  const m = String(Math.floor(secs/60)).padStart(2,'0');
  const s = String(secs%60).padStart(2,'0');
  timerEl.textContent = `${m}:${s}`;
}, 1000);
</script>
