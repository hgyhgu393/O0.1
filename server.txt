    <script>
        const mainPage = document.getElementById('main-page');
        const playerPage = document.getElementById('player-page');
        const videoElement = document.getElementById('video-element');
        const videoTitle = document.getElementById('current-video-title');
        const helper = document.getElementById('thumb-helper');
        const canvas = document.getElementById('video-canvas');

        // YouTube Audio Setup
        let ytPlayer;
        function onYouTubeIframeAPIReady() {
            ytPlayer = new YT.Player('player', {
                height: '0',
                width: '0',
                videoId: 'aj-XnqCCiNM', // ID ของเพลง เราและนาย
                events: {
                    'onReady': onPlayerReady
                }
            });
        }

        function onPlayerReady(event) {
            // พร้อมเล่นเพลง
        }

        // ระบบต้อนรับ (แบบพรีเมียม + เพลง)
        const overlay = document.getElementById('welcome-overlay');
        const text1 = document.getElementById('welcome-text-1');
        const text2 = document.getElementById('welcome-text-2');
        const msg1 = "ยินดีต้อนรับเพื่อนทุกคนนะ";
        const msg2 = "เราจะไม่ลืมกันนะ จากแทน";
        
        async function typeWriter(element, text) {
            element.classList.add('cursor-active');
            for (let i = 0; i < text.length; i++) {
                element.textContent += text.charAt(i);
                element.style.width = "auto";
                await new Promise(r => setTimeout(r, 120));
            }
            element.classList.remove('cursor-active');
        }

        window.addEventListener('load', () => {
            const hasSeenWelcome = localStorage.getItem('seen_welcome_music_v3');
            
            if (!hasSeenWelcome) {
                setTimeout(async () => {
                    overlay.classList.add('show');
                    
                    // พยายามเล่นเพลงทันที
                    if (ytPlayer && ytPlayer.playVideo) {
                        ytPlayer.playVideo();
                    }

                    await new Promise(r => setTimeout(r, 1200));
                    
                    // พิมพ์ข้อความที่ 1
                    await typeWriter(text1, msg1);
                    
                    await new Promise(r => setTimeout(r, 800));
                    
                    // พิมพ์ข้อความที่ 2
                    await typeWriter(text2, msg2);
                    
                    await new Promise(r => setTimeout(r, 2500));
                    overlay.classList.add('fade-out');
                    localStorage.setItem('seen_welcome_music_v3', 'true');
                    
                    // ค่อยๆ เบาเสียงเพลงลงหลังจากเข้าหน้าเว็บหลัก (ถ้าต้องการให้เพลงหยุด)
                    // setTimeout(() => ytPlayer.stopVideo(), 2000); 
                }, 500);
            }
        });

        // ฟังก์ชันดึงรูปจากวิดีโอมาทำหน้าปก
        function captureThumb() {
            helper.currentTime = 1;
            helper.onseeked = function() {
                canvas.width = helper.videoWidth;
                canvas.height = helper.videoHeight;
                const ctx = canvas.getContext('2d');
                ctx.drawImage(helper, 0, 0, canvas.width, canvas.height);
            }
        }
        
        helper.addEventListener('loadedmetadata', captureThumb);

        function playVideo(title, url) {
            // ถ้าเล่นวิดีโอหลัก ให้หยุดเพลงเบื้องหลัง
            if (ytPlayer && ytPlayer.pauseVideo) {
                ytPlayer.pauseVideo();
            }
            
            mainPage.classList.add('hidden');
            playerPage.classList.remove('hidden');
            videoTitle.innerText = title;
            videoElement.src = url;
            videoElement.play();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function showMainPage() {
            videoElement.pause();
            videoElement.src = "";
            playerPage.classList.add('hidden');
            mainPage.classList.remove('hidden');
        }
    </script>
